// Add to school_model.dart

class PMColourBand {
  final String name;         // "Orange"
  final String hexColor;     // "#F97316"
  final int levelStart;      // 13
  final int levelEnd;        // 14
  final String yearLevel;    // "Year 1 (End)"
  final String stage;        // "Transitional"
  final String description;  // "Building reading stamina..."

  const PMColourBand({
    required this.name,
    required this.hexColor,
    required this.levelStart,
    required this.levelEnd,
    required this.yearLevel,
    required this.stage,
    required this.description,
  });

  // Check if a PM level falls within this band
  bool containsLevel(int level) => level >= levelStart && level <= levelEnd;
}

// Pre-configured Victorian PM Colour Bands
const List<PMColourBand> victorianPMBands = [
  PMColourBand(
    name: 'Magenta',
    hexColor: '#E91E8C',
    levelStart: 1, levelEnd: 1,
    yearLevel: 'Foundation (Start)',
    stage: 'Emergent',
    description: 'Learning print concepts and directionality',
  ),
  // ... additional bands
  PMColourBand(
    name: 'Orange',
    hexColor: '#F97316',
    levelStart: 13, levelEnd: 14,
    yearLevel: 'Year 1 (End)',
    stage: 'Transitional',
    description: 'Building reading stamina. Longer texts with chapters.',
  ),
  // ... remaining bands
];

// Helper to get colour band for a student's current level
PMColourBand? getColourBandForLevel(int pmLevel) {
  return victorianPMBands.firstWhereOrNull(
    (band) => band.containsLevel(pmLevel),
  );
}

🎨 Implementation Notes

  • Why Colour Bands Matter: Parents in Victoria see these exact colours on physical book stickers. Using them in-app creates instant recognition and trust.
  • Accessibility: Always pair colour with text labels (e.g., "Orange · Level 14") for colour-blind users.
  • Progress Visualisation: The gradient progress bar shows movement between adjacent colour bands - highly motivating for both children and parents.
  • Book Integration: When implementing ISBN scanning, cross-reference PM level from book metadata and display the corresponding colour band.
  • Teacher Override: Allow teachers to manually adjust colour band assignments if their school uses a slightly different mapping.
  • Celebration Moments: Trigger special animations when a student advances to a new colour band - these are milestone moments!
